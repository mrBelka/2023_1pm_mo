cd - сменить директорию
dir (ls) - посмотреть директорию
tab - автодополнение
.. - см. на уровень выше
mkdir - создать директорию
вверх-вниз - переключение между историей команд
cat - распечатать содержимое файла
git clone путь_до_репозитория
git status - посмотреть статус
git add . - добавить все содержимое директории к отслеживанию
git commit -m "Commit_name" - сделать коммит
git push - синхронизировать локальный и удаленный репозиторий

rm - удаление файла
rm -r - удаление директории
